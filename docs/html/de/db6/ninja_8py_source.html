101 class Target(object):
102  """Target represents the paths used within a single gyp target.
103 
104  Conceptually, building a single target A is a series of steps:
105 
106  1) actions/rules/copies generates source/resources/etc.
107  2) compiles generates .o files
108  3) link generates a binary (library/executable)
109  4) bundle merges the above in a mac bundle
110 
111  (Any of these steps can be optional.)
112 
113  From a build ordering perspective, a dependent target B could just
114  depend on the last output of this series of steps.
115 
116  But some dependent commands sometimes need to reach inside the box.
117  For example, when linking B it needs to get the path to the static
118  library generated by A.
119 
120  This object stores those paths. To keep things simple, member
121  variables only store concrete paths to single files, while methods
122  compute derived values like "the last output of the target".
123  """
124  def __init__(self, type):
125  # Gyp type ("static_library", etc.) of this target.
126  self.type = type
127  # File representing whether any input dependencies necessary for
128  # dependent actions have completed.
129  self.preaction_stamp = None
130  # File representing whether any input dependencies necessary for
131  # dependent compiles have completed.
132  self.precompile_stamp = None
133  # File representing the completion of actions/rules/copies, if any.
134  self.actions_stamp = None
135  # Path to the output of the link step, if any.
136  self.binary = None
137  # Path to the file representing the completion of building the bundle,
138  # if any.
139  self.bundle = None
140  # On Windows, incremental linking requires linking against all the .objs
141  # that compose a .lib (rather than the .lib itself). That list is stored
142  # here.
143  self.component_objs = None
144  # Windows only. The import .lib is the output of a build step, but
145  # because dependents only link against the lib (not both the lib and the
146  # dll) we keep track of the import library here.
147  self.import_lib = None

181 # A small discourse on paths as used within the Ninja build:
182 # All files we produce (both at gyp and at build time) appear in the
183 # build directory (e.g. out/Debug).
184 #
185 # Paths within a given .gyp file are always relative to the directory
186 # containing the .gyp file. Call these "gyp paths". This includes
187 # sources as well as the starting directory a given gyp rule/action
188 # expects to be run from. We call the path from the source root to
189 # the gyp file the "base directory" within the per-.gyp-file
190 # NinjaWriter code.
191 #
192 # All paths as written into the .ninja files are relative to the build
193 # directory. Call these paths "ninja paths".
194 #
195 # We translate between these two notions of paths with two helper
196 # functions:
197 #
198 # - GypPathToNinja translates a gyp path (i.e. relative to the .gyp file)
199 # into the equivalent ninja path.
200 #
201 # - GypPathToUniqueOutput translates a gyp path into a ninja path to write
202 # an output file; the result can be namespaced such that it is unique
203 # to the input file name as well as the output target name.
204

285  def GypPathToNinja(self, path, env=None):

305  def GypPathToUniqueOutput(self, path, qualified=True):
306  """Translate a gyp path to a ninja path for writing output.
307 
308  If qualified is True, qualify the resulting filename with the name
309  of the target. This is necessary when e.g. compiling the same
310  path twice for two separate output targets.
311 
312  See the above discourse on path conversions."""
313 
314  path = self.ExpandSpecial(path)
315  assert not path.startswith('$'), path
316 
317  # Translate the path following this scheme:
318  # Input: foo/bar.gyp, target targ, references baz/out.o
319  # Output: obj/foo/baz/targ.out.o (if qualified)
320  # obj/foo/baz/out.o (otherwise)
321  # (and obj.host instead of obj for cross-compiles)
322  #
323  # Why this scheme and not some other one?
324  # 1) for a given input, you can compute all derived outputs by matching
325  # its path, even if the input is brought via a gyp file with '..'.
326  # 2) simple files like libraries and stamps have a simple filename.
327 
328  obj = 'obj'
329  if self.toolset != 'target':
330  obj += '.' + self.toolset
331 
332  path_dir, path_basename = os.path.split(path)
333  assert not os.path.isabs(path_dir), (
334  "'%s' can not be absolute path (see crbug.com/462153)." % path_dir)
335 
336  if qualified:
337  path_basename = self.name + '.' + path_basename
338  return os.path.normpath(os.path.join(obj, self.base_dir, path_dir,
339  path_basename))

620  def WriteRules(self, rules, extra_sources, prebuild,
621  mac_bundle_resources, extra_mac_bundle_resources):
622  env = self.GetToolchainEnv()
623  all_outputs = []
624  for rule in rules:
625  # Skip a rule with no action and no inputs.
626  if 'action' not in rule and not rule.get('rule_sources', []):
627  continue
628 
629  # First write out a rule for the rule action.
630  name = '%s_%s' % (rule['rule_name'], self.hash_for_rules)
631 
632  args = rule['action']
633  description = self.GenerateDescription(
634  'RULE',
635  rule.get('message', None),
636  ('%s ' + generator_default_variables['RULE_INPUT_PATH']) % name)
637  is_cygwin = (self.msvs_settings.IsRuleRunUnderCygwin(rule)
638  if self.flavor == 'win' else False)
639  pool = 'console' if int(rule.get('ninja_use_console', 0)) else None
640  rule_name, args = self.WriteNewNinjaRule(
641  name, args, description, is_cygwin, env, pool)
642 
643  # TODO: if the command references the outputs directly, we should
644  # simplify it to just use $out.
645 
646  # Rules can potentially make use of some special variables which
647  # must vary per source file.
648  # Compute the list of variables we'll need to provide.
649  special_locals = ('source', 'root', 'dirname', 'ext', 'name')
650  needed_variables = set(['source'])
651  for argument in args:
652  for var in special_locals:
653  if '${%s}' % var in argument:
654  needed_variables.add(var)
655 
656  def cygwin_munge(path):
657  # pylint: disable=cell-var-from-loop
658  if is_cygwin:
659  return path.replace('\\', '/')
660  return path
661 
662  inputs = [self.GypPathToNinja(i, env) for i in rule.get('inputs', [])]
663 
664  # If there are n source files matching the rule, and m additional rule
665  # inputs, then adding 'inputs' to each build edge written below will
666  # write m * n inputs. Collapsing reduces this to m + n.
667  sources = rule.get('rule_sources', [])
668  num_inputs = len(inputs)
669  if prebuild:
670  num_inputs += 1
671  if num_inputs > 2 and len(sources) > 2:
672  inputs = [self.WriteCollapsedDependencies(
673  rule['rule_name'], inputs, order_only=prebuild)]
674  prebuild = []
675 
676  # For each source file, write an edge that generates all the outputs.
677  for source in sources:
678  source = os.path.normpath(source)
679  dirname, basename = os.path.split(source)
680  root, ext = os.path.splitext(basename)
681 
682  # Gather the list of inputs and outputs, expanding $vars if possible.
683  outputs = [self.ExpandRuleVariables(o, root, dirname,
684  source, ext, basename)
685  for o in rule['outputs']]
686 
687  if int(rule.get('process_outputs_as_sources', False)):
688  extra_sources += outputs
689 
690  was_mac_bundle_resource = source in mac_bundle_resources
691  if was_mac_bundle_resource or \
692  int(rule.get('process_outputs_as_mac_bundle_resources', False)):
693  extra_mac_bundle_resources += outputs
694  # Note: This is n_resources * n_outputs_in_rule. Put to-be-removed
695  # items in a set and remove them all in a single pass if this becomes
696  # a performance issue.
697  if was_mac_bundle_resource:
698  mac_bundle_resources.remove(source)
699 
700  extra_bindings = []
701  for var in needed_variables:
702  if var == 'root':
703  extra_bindings.append(('root', cygwin_munge(root)))
704  elif var == 'dirname':
705  # '$dirname' is a parameter to the rule action, which means
706  # it shouldn't be converted to a Ninja path. But we don't
707  # want $!PRODUCT_DIR in there either.
708  dirname_expanded = self.ExpandSpecial(dirname, self.base_to_build)
709  extra_bindings.append(('dirname', cygwin_munge(dirname_expanded)))
710  elif var == 'source':
711  # '$source' is a parameter to the rule action, which means
712  # it shouldn't be converted to a Ninja path. But we don't
713  # want $!PRODUCT_DIR in there either.
714  source_expanded = self.ExpandSpecial(source, self.base_to_build)
715  extra_bindings.append(('source', cygwin_munge(source_expanded)))
716  elif var == 'ext':
717  extra_bindings.append(('ext', ext))
718  elif var == 'name':
719  extra_bindings.append(('name', cygwin_munge(basename)))
720  else:
721  assert var == None, repr(var)
722 
723  outputs = [self.GypPathToNinja(o, env) for o in outputs]
724  if self.flavor == 'win':
725  # WriteNewNinjaRule uses unique_name for creating an rsp file on win.
726  extra_bindings.append(('unique_name',
727  hashlib.md5(outputs[0]).hexdigest()))
728  self.ninja.build(outputs, rule_name, self.GypPathToNinja(source),
729  implicit=inputs,
730  order_only=prebuild,
731  variables=extra_bindings)
732 
733  all_outputs.extend(outputs)
734 
735  return all_outputs
