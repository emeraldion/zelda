236  HSL rgb_to_hsl(double r, double g, double b)
237  {
238 
239  // Algorithm from http://en.wikipedia.org/wiki/wHSL_and_HSV#Conversion_from_RGB_to_HSL_or_HSV
240  r /= 255.0; g /= 255.0; b /= 255.0;
241 
242  double max = std::max(r, std::max(g, b));
243  double min = std::min(r, std::min(g, b));
244  double del = max - min;
245 
246  double h = 0, s = 0, l = (max + min) / 2.0;
247 
248  if (max == min) {
249  h = s = 0; // achromatic
250  }
251  else {
252  if (l < 0.5) s = del / (max + min);
253  else s = del / (2.0 - max - min);
254 
255  if (r == max) h = (g - b) / del + (g < b ? 6 : 0);
256  else if (g == max) h = (b - r) / del + 2;
257  else if (b == max) h = (r - g) / del + 4;
258  }
259 
260  HSL hsl_struct;
261  hsl_struct.h = h / 6 * 360;
262  hsl_struct.s = s * 100;
263  hsl_struct.l = l * 100;
264 
265  return hsl_struct;
266  }

270  if (h < 0) h += 1;

279  {
280  h /= 360.0;
281  s /= 100.0;
282  l /= 100.0;
283 
284  if (l < 0) l = 0;
285  if (s < 0) s = 0;
286  if (l > 1) l = 1;
287  if (s > 1) s = 1;
288  while (h < 0) h += 1;
289  while (h > 1) h -= 1;

292  double m2;

298  double g = (h_to_rgb(m1, m2, h) * 255.0);
299  double b = (h_to_rgb(m1, m2, h-1.0/3.0) * 255.0);
