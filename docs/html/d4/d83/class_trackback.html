auto_discovery (   $text)

Search text for links, and searches links for trackback URLs.

<?php

include('trackback_cls.php'); $trackback = new Trackback('BLOGish', 'Ran Aroussi', 'UTF-8');

if ($tb_array = $trackback->auto_discovery(string TEXT)) { // Found trackbacks in TEXT. Looping... foreach($tb_array as $tb_key => $tb_url) { // Attempt to ping each one... if ($trackback->ping($tb_url, string URL, [string TITLE], [string excerpt])) { // Successful ping... echo "Trackback sent to <i>$tb_url</i>...\n"; } else { // Error pinging... echo "Trackback to <i>$tb_url</i> failed....\n"; } } } else { // No trackbacks in TEXT... echo "No trackbacks were auto-discover...\n" } ?>
